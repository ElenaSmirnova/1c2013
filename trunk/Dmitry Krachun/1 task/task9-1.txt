#include <iostream>
# include <stdlib.h>
using namespace std;

int a[1000];
int n;

int step(int left, int right, int value)
{
   if(right - left < 1) return -1;
   int c = left;
   for(int i = left; i <= right; i++){
           if(a[i] <= value){
               swap(a[i], a[c]);
               c++;
           }
   }
   swap(a[c - 1], a[left]);
   return c - 1;
}

void quicksort(int i, int j)
{
   int k = step(i, j, a[i]);
   if(k >= 0){
       quicksort(i, k - 1);
       quicksort(k + 1, j);
   }
}

int main()
{
   cin >> n;
   for(int i = 0; i < n; i++){
       cin >> a[i];
       //a[i] = rand() % 100;
   }
   quicksort(0, n - 1);
   for(int i = 0; i < n; i++) {
           cout << a[i] << " ";
   };
   return 0;
}
