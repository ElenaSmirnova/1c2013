#include <iostream>
#include<string>
using namespace std;

int main()
{
   string S;
   string S_1;
   cin >> S;
   cin >> S_1;
   int ans = 0, n, m;
   n = S.length();
   m = S_1.length();
   if(n < m){
       cout << 0;
       return 0;
   }

   for(int i = 0; i < n - m + 1; i++){
       if(S_1 == S.substr(i,m)){
           ans++;
       }
   }
   cout << ans;
   return 0;
}
