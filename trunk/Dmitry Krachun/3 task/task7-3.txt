#include <cstdio>
#include <iostream>
using namespace std;

struct tree//binary tree
{
    tree *left, *right;
    int num, cnt;//root contain cnt equal numbers num;
};
void go(tree *t)
{
    if(t){
        go(t->left);
        cout << t->num  << ":" << t->cnt << endl;
        go(t->right);
    }
}
void insert(tree *&t, int x)//insert element x in tree t;
{
    if(t == 0){
        t = new tree();
        t->left = 0;
        t->right = 0;
        t->num = x;
        t->cnt = 1;
    } else if(t->num == x) t->cnt++;
      else if(t->num > x) insert(t->left, x);
      else if(t->num < x) insert(t->right, x);
}
void delete_tree(tree *t)
{
    if(t)
    {
       if(t->left) delete_tree(t->left);
       if(t->right) delete_tree(t->right);
       delete t;
    }
}
void delete_subtree(tree *t, int x)
{
    if((t == 0) || (t->num == x)) {delete_tree(t); t = 0;}
    else if(t->num > x) delete_subtree(t->left, x);
    else delete_subtree(t->right, x);
}
int main()
{
  tree *t = 0;
  insert(t, 1);
  insert(t, 2);
  insert(t, 3);
  insert(t, 5);
  insert(t, 3);
  insert(t, 6);
  go(t);
  delete_subtree(t, 6);
}
