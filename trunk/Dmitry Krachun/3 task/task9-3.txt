#include <iostream>
using namespace std;

int n = 0, a = 0, b = 0;//number of A letters and B letters;
char str[100];
void print(int k)
{
    for(int i = 0; i < n; i++){
        if((k & (1 << i))>0) cout << 'A'; else cout << 'B';
    }
    cout << endl;
}

void write()
{
    for(int k = 0; k < (1 << n); k++){
        int s = 0;
        for(int j = 0; j < n; j++) s += ((k & (1 << j)) > 0);
        if (2 * s == n) print(k);
    }
}

int main()
{
    cin >> n;
    write();
    return 0;
}
