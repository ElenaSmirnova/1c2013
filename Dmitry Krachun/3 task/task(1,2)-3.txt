#include <cstdio>
#include <iostream>
using namespace std;

struct tree//binary tree
{
    tree *left, *right;
    int num, cnt;//root contain cnt equal numbers num;
};
void go(tree *t)
{
    if(t){
        cout << t->num  << ":" << t->cnt << endl;
        go(t->left);
        go(t->right);
    }
}
void insert(tree *&t, int x)//insert element x in tree t;
{
    if(t == 0){
        t = new tree();
        t->left = 0;
        t->right = 0;
        t->num = x;
        t->cnt = 1;
    } else {
        if(t->num == x) t->cnt++;
        if(t->num > x) insert(t->left, x);
        if(t->num < x) insert(t->right, x);
    }
}

int main()
{
  tree *t = 0;
  for(int i = 0; i < 8; i++){
    insert(t, i);
  }
  for(int i = 4; i < 11; i++){
    insert(t, i);
  }
  go(t);
}
