#include <cstdio>
#include <iostream>
#include <string>

using namespace std;
char stack[1000];
char S[100];
int n = 0, top = 0;
bool pb(char c)
{
  if((stack[top] == '(') && (c == ')')) top--;
  else stack[++top] = c;
  if(stack[top] == ')') return false; else return true;
}
int check()
{
    int i = 0;
    for(i = 0; S[i]; i++){
        if(!(pb(S[i]))) return i + 1;
    };
    return -1;
}
int main()
{
    scanf("%s", S);
    cout << check() << endl;
    return 0;
}
