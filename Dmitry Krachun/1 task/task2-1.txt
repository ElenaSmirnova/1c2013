#include <iostream>

using namespace std;

int main()
{
   int m[1000];
   int inf=1e9;
   int maxx, minn, c , i, j, n;
   cin >> n;
   for(i = 0; i < n; i++) cin >> m[i];
   maxx = -inf;
   for(i = 0; i < n; i++) if(m[i] > maxx) {maxx = m[i];};
   minn = inf;
   for(i = 0; i < n; i++) if(m[i] < minn) {minn = m[i];};
   for(c = minn; c <= maxx; c++){
       for(i = 0; i < n; i++){
           if(m[i] == c){cout << c << ' ';}
       }
   }
   cout << endl;
   for(i = 0; i < n; i++) {
       for(j = i + 1; j < n; j++) {
           if(m[i] > m[j]) {
               m[i] = m[i] + m[j];
               m[j] = m[i] - m[j];
               m[i] = m[i] - m[j];
           }
       }
   }
   for (i = 0; i < n; i++){
       cout << m[i] << ' ';
   }

   return 0;

}
