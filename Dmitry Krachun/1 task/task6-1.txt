#include <iostream>

using namespace std;

int main()
{
   int n, i, j;
   bool prime[100000];

   cin >> n;
   for(i = 0; i < n; i++){
       prime[i]=1;
   }
   prime[1]=0;
   for (i = 2; i < n; i++){
       if(prime[i]){
           for(j = 2; i * j <= n; j++){
               prime[i*j] = 0;
           }
       }
   }
   for(i = 1; i <= n; i++){
       if(prime[i]){
           cout << i << " ";
       }
   }
   return 0;
}
